% --------------------------------------------------
% glossary.tex -- term glossary for the fl-trilemma report
% --------------------------------------------------

% ----- FL fundamentals -----
\newglossaryentry{globalmodel}{
	name={Global model},
	description={Shared model that is broadcast to every client, each round}
}
\newglossaryentry{localmodel}{
	name={Local model},
	description={Client's copy of the global model after training}
}
\newglossaryentry{communicationround}{
	name={Communication round},
	description={One FL iteration}
}
\newglossaryentry{localsgdstep}{
	name={Local (SGD) step},
	description={One batch gradient update during local training}
}
\newglossaryentry{localepoch}{
	name={Local epoch},
	description={One full pass over the clients local dataset during training}
}

% ----- Privacy -----
\newglossaryentry{localdp}{
	name={Local DP},
	description={Each client perturbs its own update with noise before transmission}
}
\newglossaryentry{centraldp}{
	name={Central DP},
	description={A trusted aggregator adds noise once after collecting raw updates}
}
\newglossaryentry{privacybudget}{
	name={Privacy budget},
	description={The acceptable privacy loss}
}

% ----- Robustness / Byzantine / FLTrust -----
\newglossaryentry{byzantineadversary}{
	name={Byzantine adversary},
	description={A client that can deviate arbitrarily from the training protocol}
}
\newglossaryentry{byzantinefraction}{
	name={Byzantine fraction},
	description={The share of clients in a run that are malicious}
}
\newglossaryentry{honestclient}{
	name={Honest client},
	description={A client that follows the training protocol}
}
\newglossaryentry{maliciousclient}{
	name={Malicious client},
	description={A byzantine adversary that tries to corrupt the global model}
}
\newglossaryentry{trustscore}{
	name={Trust score},
	description={FLTrust's per-client aggregation weight}
}
\newglossaryentry{rootdataset}{
	name={Root dataset},
	description={A class-balanced dataset held by the server to compute the trust scores}
}
\newglossaryentry{updateanchor}{
	name={Update anchor},
	description={A reference update, trained by the server to compute the trust scores}
}
\newglossaryentry{labelflippingattack}{
	name={Label-flipping attack},
	description={A byzantine attack that swaps two labels during training}
}
\newglossaryentry{attackscale}{
	name={Attack scale},
	description={A multiplier applied to a malicious client's resulting update to amplify its effect on aggregation}
}

% ----- Efficiency -----
\newglossaryentry{sparsificationratio}{
	name={Sparsification ratio},
	description={The fraction of parameters retained by Top-k compression}
}

% ----- Trilemma metrics -----
\newglossaryentry{accuracygap}{
	name={Accuracy gap},
	description={The drop in final accuracy an attacked configuration suffers, relative to a clean FedAvg baseline}
}

% ----- Algorithms -----
\newglossaryentry{fedavg}{
	name={FedAvg},
	description={A weighted average of client updates by local dataset size}
}
