\begin{frame}{Secure Aggregation Protocol by Bonawitz et al. \cite{bonawitz2016practicalsecureaggregationfederated}}
\begin{columns}[T]
    
    \begin{column}{0.5 \textwidth}
        \textbf{Protocol Summary}
        \begin{itemize}
            \item Uses masking vectors to encrypt individual gradients that cancel out in the aggregation
            \item Robust to user dropout by using secret sharing and a second masking vector
            \item Secure against malicious adversaries with limited memory access by using OneTime Pads and a sufficiently high threshold for secret reconstruction
        \end{itemize}
    \end{column}
    \begin{column}{0.5\textwidth}

\begin{frame}{SAFELearn: Secure Aggregation for private Federated Learning \cite{cryptoeprint:2021/386}}
    \begin{columns}
        \begin{column}{0.65\textwidth}
            \textbf{2 Main Variants:}\\
            \begin{itemize}
                \item Fully Homomorphic Encryption (FHE)
                \item [] \textcolor{gray}{Uses additively homomorphic encryption scheme}
                %\item [] \textcolor{gray}{Encryption key is split among clients using secret sharing}
                \begin{enumerate}
                    \item Clients send encrypted gradients to the server
                    \item Server aggregates encrypted gradients and shares result
                    \item Clients jointly decrypt the updated model
                \end{enumerate}
                \pause
                \item Secure Two / Multi-Party Computation (2PC/MPC)
                \begin{enumerate}
                    \item Client(s) split updates among servers using secret sharing
                    \item Servers jointly aggregate and distribute updates
                    \item Client(s) reconstruct the result
                \end{enumerate}
            \end{itemize}
        \end{column}
        \begin{column}{0.35\textwidth}
            \pause
            \begin{itemize}
                \item[\textcolor{green}{+}] Privacy preservation against malicious servers
                \item[\textcolor{red}{!}] Assumes honest clients
                \item[] High communication costs
                \item[] MPC/2PC variant assumes non-colluding servers
            \end{itemize}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{Robust Aggregation}
    
\begin{alertblock}{Motivation}
    The aggregation protocols should be robust against malicious clients trying to manipulate the training process.
\end{alertblock}
\begin{columns}
\begin{column}{0.5\textwidth}
\textbf{Main Threats}
\begin{itemize}
    \item Poisoning Attacks
    \begin{itemize}
        \item \textbf{Backdoor:} Implement sensitivity to a specific trigger
        \item \textbf{Untargeted:} Degrade overall model performance
        \item \textbf{Targeted:} Degrade performance on apecific class(es)
    \end{itemize}
\end{itemize} 
\end{column}
\begin{column}{0.5\textwidth}
    \textbf{Defense Mechanisms}
    \begin{itemize}
        \item Robust aggregation rules
        \begin{itemize}
            \item Krum
            \item Trimmed Mean
            \item FLTrust
        \end{itemize}
        \item Anomaly detection on client updates
    \end{itemize}
\end{column}
\end{columns}
\end{frame}

\subsection{Robust Aggregation Rules}
\begin{frame}{Robust Aggregation Rules}
    \begin{itemize}
        \item \textbf{Krum \cite{Krum}:} Only consider updates that are closest to other updates
        \item[\textcolor{red}{!}] \textcolor{black}{Assumes few malicious clients and that honest updates cluster together $\rightarrow$ fails for non-IID data}
        \item \textbf{Trimmed Mean \cite{TrimmedMean}:} For each coordinate of the update vector, cut the top and bottom n\% before averaging
        \item[\textcolor{red}{!}] \textcolor{black}{neglects correlations between dimensions \\ for fixed, known thresholds, malicious updates can be crafted accordingly}
        \item \textbf{FLTrust \cite{cao2021fltrust}:} Maintain small root dataset on server to calculate update anchor, weigh client updates by similarity to anchor
        \item[\textcolor{red}{!}] \textcolor{black}{Assumes trusted, representative dataset available}
        \pause
        \item[] \textbf{Under highly non-IID data, honest clients naturally produce very different updates, making it hard to distinguish them from dishonest ones}
    \end{itemize}
\end{frame}

\subsection{Trust and Verification}
\begin{frame}{Trust and Verification}
    \begin{itemize}
        \item \textbf{Zero-Knowledge-Proofs} to prove that updates satisfy certain requirements \cite{burkhalter2023rofl}
        \item[\textcolor{red}{!}] Computationally expensive for large updates
        \item \textbf{Verifiable Computation:}  Client proves it ran the training algorithm correctly, not just that the output has certain properties
        \item[\textcolor{red}{!}] Proof generation cost grows exponentially with number of gradient steps
        \item \textbf{Auditing/Attestation Frameworks:} audit logs + secure hardware attestation instead of cryptographic proofs
        \item[\textcolor{red}{!}] Requires legal accountability
    \end{itemize}
\end{frame}

\section{Privacy-Robustness-Utility Tradeoff}
\begin{frame}{Privacy-Robustness-Utility Tradeoff \cite{allouah2023trilemma}}
\begin{itemize}
    \item Meaningful privacy guarantees introduce noise reducing model performance
    \item Robust aggregation requires access to individual updates instead of averaged scores
    \item Robust aggregation on non-IID data will likely discard legitimate updates; trimming or median operations introduce bias
\end{itemize}
    
\end{frame}

\begin{frame}{Secure Aggregation Protocol by Bonawitz et al. \cite{bonawitz2016practicalsecureaggregationfederated}}
    \begin{columns}
        \begin{column}{0.5\textwidth}
            %protocol steps
            \begin{enumerate}
                \item Users pairwise agree on input perturbation pair $p_{u,v}, p_{v,u}$
                \item Each user masks their gradient using the sum of all perturbations known and sends the result to the Server
                \item The Server aggregates as usual because the perturbations are structured to cancel each other
            \end{enumerate}
        \end{column}
        \pause
        \begin{column}{0.5\textwidth}
            %up- and downsides
            \begin{itemize}
                \item[\textbf{\textcolor{green}{+}}] information theoretically secure through OneTime Pads, even for malicious adversaries with limited memory access
                \pause
                \item[\textbf{\textcolor{red}{!}}] strong assumptions:
                \begin{itemize}
                    \item[] pairwise secure channels with sufficient bandwidth for secret exchange  
                    \item[] protocol completion by all users
                \end{itemize}
            \end{itemize}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{Secure Aggregation Protocol by Bonawitz et al. \cite{bonawitz2016practicalsecureaggregationfederated}}
    \begin{columns}
        \begin{column}{0.6\textwidth}
            %protocol steps
            \begin{enumerate}
            \setcounter{enumi}{-1}
                \item Each user creates a key pair and broadcasts their public key to all other users

                \item \textcolor{gray}{Perturbation vector agreement}
                \begin{enumerate}
                    \item Secret sharing: Users distribute perturbation vector among all other users with reconstruction threshold $t$
                \end{enumerate}
                
                \item \textcolor{gray}{Masking + sending of gradients}
                \begin{enumerate}
                    \item Only use perturbations shared between surviving users. Reveal secrets of dropped out users if at least $t$ users survived
                \end{enumerate}
                
                \item \textcolor{gray}{Aggregation by server
                }
            \end{enumerate}
        \end{column}
        \pause
        \begin{column}{0.4\textwidth}
            %up- and downsides
            \begin{itemize}
                \item[\textbf{\textcolor{green}{+}}] Robustness to user dropout
                \pause
                \item[\textbf{\textcolor{red}{!}}] Secret exposure in case of malicious server or delayed update sending; increased communication cost
            \end{itemize}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{Secure Aggregation Protocol by Bonawitz et al. \cite{bonawitz2016practicalsecureaggregationfederated}}
    \begin{columns}
        \begin{column}{0.5\textwidth}
            %protocol steps
            \begin{enumerate}
            \setcounter{enumi}{-1}
                \item \textcolor{gray}{Key pair generation} + generation of masking vector $b$
                \item \textcolor{gray}{Perturbation vector agreement}
                \begin{enumerate}
                    \item \textcolor{gray}{Secret sharing} including $b$
                \end{enumerate}
                \item \textcolor{gray}{Masking + sending of gradients:} Use $b$ as second mask
                \begin{enumerate}
                    \item \textcolor{gray}{Reveal secrets of dropped out users if at least $t$ users survived:} Each user will only reveal a share of $p_u$ OR $b_u$
                \end{enumerate}
                \item \textcolor{gray}{Aggregation by server} 
                
            \end{enumerate}
        \end{column}
        \pause
        \begin{column}{0.5\textwidth}
            %up- and downsides
            \begin{itemize}
                \item[\textbf{\textcolor{green}{+}}] For $t > \frac{n}{2}$, all gradients stay masked by either $p$ or $b$; extend to stronger adversaries by increasing threshold
                \pause
                \item[\textbf{\textcolor{red}{!}}] Communication cost of $O(kn^2)$
            \end{itemize}
            \pause
            \begin{alertblock}{Reducing Complexity}
                    Share seeds for PRG instead of full vectors using DH Key Exchange via server
            \end{alertblock}
            \pause
            \begin{alertblock}{Getting rid of secured channel assumption}
                Setup secured channels via server-mediated DH  $\rightarrow$ adds PFS
            \end{alertblock}
        \end{column}
    \end{columns}
